`CapabilityRequest`, fix the named-pipe SDDL gap so Low Integrity children can actually connect,

capabilities. This plan delivers the end-to-end wire; Plan 11-02 adds the interactive
approval device branch and the test coverage.

Output: A working (but uncovered) capability pipe. A sandboxed child can send a
CapabilityRequest and the supervisor handler validates + dispatches it.

   fn start_capability_pipe_server(&self) -> Result<()> {
       let session_token = self.session_token.clone();
       let rendezvous_path = self.cap_pipe_rendezvous_path.clone();
       let audit_tx = self.audit_tx.clone();
       let child_handle_slot = self.child_process_for_broker.clone();
       let terminate_requested = self.terminate_requested.clone();
       let approval_backend = self.approval_backend.clone(); // Arc<dyn ApprovalBackend>

       std::thread::spawn(move || {
           let mut sock = match nono::SupervisorSocket::bind_low_integrity(&rendezvous_path) {
               Ok(s) => s,
               Err(e) => {
                   tracing::error!(target: "nono::supervisor", error = %e, "capability pipe bind failed");
                   return;
               }
           };

           // Block until the child process handle is available (spawn happens after initialize()).
           let child_handle: windows_sys::Win32::Foundation::HANDLE = loop {
               if terminate_requested.load(std::sync::atomic::Ordering::SeqCst) { return; }
               {
                   let slot = child_handle_slot.lock().expect("child_handle_slot poisoned");
                   if let Some(h) = *slot { break h; }
               }
               std::thread::sleep(std::time::Duration::from_millis(25));
           };
           let target = nono::BrokerTargetProcess::from_handle(child_handle);

           let mut seen_request_ids = std::collections::HashSet::new();
           let mut local_audit: Vec<AuditEntry> = Vec::new();
           loop {
               if terminate_requested.load(std::sync::atomic::Ordering::SeqCst) { return; }
               let msg = match sock.recv_message() {
                   Ok(m) => m,
                   Err(e) => {
                       tracing::debug!(target: "nono::supervisor", error = %e, "capability pipe recv ended");
                       break;

                   &mut sock,
                   msg,
                   approval_backend.as_ref(),
                   target.clone(),
                   &mut seen_request_ids,
                   &mut local_audit,
                   &session_token,

                   tracing::warn!(target: "nono::supervisor", error = %e, "capability handler error");
               }
               if !local_audit.is_empty() {
                   let drained: Vec<AuditEntry> = local_audit.drain(..).collect();
                   let _ = audit_tx.send(drained);
               }
           }
       });
       Ok(())
   }
