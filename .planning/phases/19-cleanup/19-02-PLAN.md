```rust
/// Process-global lock; acquire before any env-var mutation in a test.
pub static ENV_LOCK: std::sync::Mutex<()>;
pub fn lock_env() -> std::sync::MutexGuard<'static, ()>;

/// RAII guard: set env vars, capture originals, restore on drop.
pub struct EnvVarGuard { /* ... */ }
impl EnvVarGuard {
    pub fn set_all(vars: &[(&'static str, &str)]) -> Self;
    pub fn remove(&self, key: &str);
}
```

Canonical usage pattern (copied from `profile/builtin.rs::test_all_profiles_signal_mode_resolves` call site area):
```rust
#[test]
fn some_env_sensitive_test() {
    let _lock = crate::test_env::lock_env();
    let _env = crate::test_env::EnvVarGuard::set_all(&[
        ("HOME", "/home/nono-test"),
        ("USERPROFILE", "C:\\nono-test"),          // Windows analog of HOME
        ("XDG_CONFIG_HOME", "/home/nono-test/.config"),
    ]);
    // ... test body that indirectly reads HOME/USERPROFILE via validated_home(), etc.
}
```

Note: `lock_env()` is MANDATORY even if the test only READS env vars — because other tests WRITE them in parallel and the lock is the only serialization point. `EnvVarGuard` alone is insufficient without the lock.
