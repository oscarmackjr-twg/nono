pub enum HandleKind {
    File = 0, Socket = 1, Pipe = 2, JobObject = 3, Event = 4, Mutex = 5,

pub enum SocketRole { Connect, Bind, Listen }
pub enum PipeDirection { Read, Write, ReadWrite }

#[serde(tag = "type")]
pub enum HandleTarget {
    FilePath { path: PathBuf },
    SocketEndpoint { protocol: SocketProtocol, host: String, port: u16, role: SocketRole },
    PipeName { name: String },
    JobObjectName { name: String },
    EventName { name: String },
    MutexName { name: String },
}

pub struct CapabilityRequest {
    pub request_id: String,
    pub path: Option<PathBuf>,           // deprecated in place per 18-01 D-01
    pub kind: HandleKind,                // added in 18-01
    pub target: HandleTarget,            // added in 18-01
    pub access_mask: u32,                // added in 18-01
    pub access: AccessMode,              // existing Phase 11 field — preserved for File path
    pub reason: Option<String>,
    pub child_pid: u32,
    pub session_id: String,
    pub session_token: String,
}

pub enum ResourceTransferKind {
    SidebandFileDescriptor,
    DuplicatedWindowsHandle,
    SocketProtocolInfoBlob,              // added in Plan 18-01; constructor in 18-02
}

pub struct ResourceGrant {
    pub transfer: ResourceTransferKind,
    pub resource_kind: GrantedResourceKind,
    pub access: AccessMode,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub raw_handle: Option<u64>,         // carries duplicated HANDLE for File/Pipe/Job/Event/Mutex
    #[serde(skip_serializing_if = "Option::is_none")]
    pub protocol_info_blob: Option<Vec<u8>>,  // carries WSAPROTOCOL_INFOW for Socket
}

pub enum SupervisorMessage {
    Request(CapabilityRequest),
    OpenUrl(UrlOpenRequest),
    Terminate { session_id: String },
    Detach { session_id: String },
}

pub enum SupervisorResponse {
    Decision {
        request_id: String,
        decision: ApprovalDecision,
        grant: Option<ResourceGrant>,
    },
    UrlOpened { request_id: String, success: bool, error: Option<String> },
}

pub enum ApprovalDecision {
    Granted,
    Denied { reason: String },
    Timeout,
}

       Method signatures (verbatim from CONTEXT.md D-08, with `&mut SupervisorSocket`
       added as the transport argument per 18-RESEARCH lines 675-679 — the caller owns

       ```rust
       #[cfg(target_os = "windows")]
       pub fn request_socket(
           cap_pipe: &mut SupervisorSocket,
           host: &str,
           port: u16,
           protocol: SocketProtocol,
           role: SocketRole,
           access_mask: u32,
           reason: Option<&str>,
       ) -> Result<RawSocket> {
           let target = HandleTarget::SocketEndpoint {
               protocol,
               host: host.to_string(),
               port,
               role,
           };
           let grant = send_capability_request(
               cap_pipe,
               HandleKind::Socket,
               target,
               access_mask,
               reason,
           )?;
           // Validate the supervisor's grant matches the Socket transport contract
           // (defense-in-depth: the supervisor is supposed to enforce this server-side
           // per Plan 18-02, but we double-check on the client side so a corrupted
           // response doesn't get silently misinterpreted as a HANDLE).
           if grant.transfer != ResourceTransferKind::SocketProtocolInfoBlob {
               return Err(NonoError::SandboxInit(format!(
                   "expected SocketProtocolInfoBlob transfer for Socket grant, \
                    got {:?}", grant.transfer
               )));
           }
           let blob = grant.protocol_info_blob.ok_or_else(|| {
               NonoError::SandboxInit(
                   "Socket grant missing protocol_info_blob".to_string()
               )
           })?;
           reconstruct_socket_from_blob(&blob)
       }

       #[cfg(not(target_os = "windows"))]
       pub fn request_socket(
           _cap_pipe: &mut SupervisorSocket,
           _host: &str,
           _port: u16,
           _protocol: SocketProtocol,
           _role: SocketRole,
           _access_mask: u32,
           _reason: Option<&str>,
       ) -> Result<RawSocket> {
           Err(NonoError::UnsupportedPlatform(
               unsupported_platform_message().to_string()
           ))
       }
       ```

       /// Construct a CapabilityRequest, send it, wait for response, demultiplex
       /// the Decision into a typed Result. Shared by all 5 Windows-arm SDK methods.
       #[cfg(target_os = "windows")]
       fn send_capability_request(
           cap_pipe: &mut SupervisorSocket,
           kind: HandleKind,
           target: HandleTarget,
           access_mask: u32,
           reason: Option<&str>,
       ) -> Result<ResourceGrant> {
           // Generate a UUID-shaped request_id. Reuse uuid crate (already a workspace
           // dep per Phase 11). If uuid is not in the nono crate's deps, use
           // getrandom + hex (already in workspace per 18-RESEARCH line 759).
           let request_id = generate_request_id();

           // Read NONO_SESSION_TOKEN from env. Phase 11 D-01 contract: hex-encoded
           // 32 bytes, set by execution_runtime.rs:224 on child spawn. Missing →
           // fail-secure with a clear error (the supervisor would reject the
           // request anyway via constant-time token check, but failing fast on
           // the client side gives a better diagnostic).
           let session_token = std::env::var("NONO_SESSION_TOKEN")
               .map_err(|_| NonoError::SandboxInit(
                   "NONO_SESSION_TOKEN env var not set; AIPC SDK requires \
                    Phase 11 supervisor plumbing".to_string()
               ))?;

           // Read session_id from env (set by Phase 11 plumbing alongside the token).
           // The supervisor uses session_id only for audit correlation; not
           // security-critical (token is the auth credential).
           let session_id = std::env::var("NONO_SESSION_ID").unwrap_or_default();

           // Read child_pid via std::process::id() — the supervisor uses this for
           // audit-log correlation only (server-side validation is via the named
           // pipe's PID-binding, not via this field).
           let child_pid = std::process::id();

           let req = CapabilityRequest {
               request_id: request_id.clone(),
               // Phase 11 path field deprecated in place per 18-01 D-01;
               // set to None for all AIPC requests since target carries the typed payload.
               path: None,
               kind,
               target,
               access_mask,
               // Phase 11 access field preserved for File path; for AIPC kinds
               // the access information lives in access_mask. Use AccessMode::Read
               // as a benign default (the supervisor ignores this field for non-File
               // kinds per Plan 18-01 dispatch).
               access: crate::capability::AccessMode::Read,
               reason: reason.map(str::to_string),
               child_pid,
               session_id,
               session_token,
           };

           cap_pipe.send_message(&SupervisorMessage::Request(req))?;

           match cap_pipe.recv_response()? {
               SupervisorResponse::Decision { request_id: resp_id, decision, grant } => {
                   if resp_id != request_id {
                       return Err(NonoError::SandboxInit(format!(
                           "supervisor response request_id mismatch: \
                            expected {request_id}, got {resp_id}"
                       )));
                   }
                   match decision {
                       ApprovalDecision::Granted => {
                           grant.ok_or_else(|| NonoError::SandboxInit(
                               "supervisor granted but returned no ResourceGrant"
                                   .to_string()
                           ))
                       }
                       ApprovalDecision::Denied { reason } => {
                           Err(NonoError::SandboxInit(format!(
                               "supervisor denied capability: {reason}"
                           )))
                       }
                       ApprovalDecision::Timeout => {
                           Err(NonoError::SandboxInit(
                               "supervisor approval timed out".to_string()
                           ))
                       }
                   }
               }
               other => Err(NonoError::SandboxInit(format!(
                   "expected Decision response, got {other:?}"
               ))),
           }
       }

       #[cfg(target_os = "windows")]
       fn generate_request_id() -> String {

           // research environment availability table.

       For `request_socket`, add the WSASocketW reconstruction helper below
       `send_capability_request`. This is the ONE place in the SDK where `unsafe` is
       needed; scope it tightly with a `// SAFETY:` comment per project standards:

       ```rust
       #[cfg(target_os = "windows")]
       fn reconstruct_socket_from_blob(blob: &[u8]) -> Result<RawSocket> {
           use std::mem::size_of;
           use windows_sys::Win32::Networking::WinSock::{
               WSASocketW, WSAPROTOCOL_INFOW, FROM_PROTOCOL_INFO, INVALID_SOCKET,
               WSA_FLAG_OVERLAPPED, AF_UNSPEC, WSAGetLastError,
           };
           if blob.len() != size_of::<WSAPROTOCOL_INFOW>() {
               return Err(NonoError::SandboxInit(format!(
                   "WSAPROTOCOL_INFOW blob length mismatch: expected {} bytes, \
                    got {} bytes (likely truncated transport)",
                   size_of::<WSAPROTOCOL_INFOW>(), blob.len()
               )));
           }
           // SAFETY: blob is a Vec<u8> of exactly size_of::<WSAPROTOCOL_INFOW>() bytes
           // (validated above). The WSAPROTOCOL_INFOW struct is `#[repr(C)]` and has
           // no padding bytes that matter for WSASocketW's interpretation. The blob
           // was produced by WSADuplicateSocketW server-side per Plan 18-02 and
           // serialized as raw bytes; we deserialize back to the same struct shape.
           let proto_info: WSAPROTOCOL_INFOW = unsafe {
               std::ptr::read_unaligned(blob.as_ptr() as *const WSAPROTOCOL_INFOW)
           };
           // SAFETY: WSASocketW with FROM_PROTOCOL_INFO and a non-null pointer to a
           // valid WSAPROTOCOL_INFOW reconstructs the duplicated socket. The blob
           // is single-use and target-PID-bound (per Plan 18-02 lifecycle); calling
           // on the wrong PID returns INVALID_SOCKET with WSAEINVAL.
           let sock = unsafe {
               WSASocketW(
                   AF_UNSPEC as i32, 0, 0,
                   &proto_info as *const _,
                   0,
                   WSA_FLAG_OVERLAPPED,
               )
           };
           if sock == INVALID_SOCKET {
               // SAFETY: WSAGetLastError is thread-local Win32 state; safe to call
               // immediately after a Winsock failure to retrieve the error code.
               let err = unsafe { WSAGetLastError() };
               return Err(NonoError::SandboxInit(format!(
                   "WSASocketW(FROM_PROTOCOL_INFO) failed: WSAGetLastError = {err}"
               )));
           }
           Ok(sock as RawSocket)
       }
       ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| child SDK (untrusted code) → child SDK module | The `request_*` functions are in the same `crates/nono` library as the supervisor types, so technically same-process. However, the SDK is the SHAPE the child code uses to construct requests — a malicious SDK author could stamp arbitrary `kind`/`target`/`access_mask` values on a `CapabilityRequest` directly, bypassing the SDK helpers. The SDK is convenience-only; the supervisor remains the load-bearing validator (CONTEXT.md D-07). This boundary's threat model is therefore "the SDK is helpful but not security-critical". |
| child process env (NONO_SESSION_TOKEN, NONO_SUPERVISOR_PIPE) → SDK | The SDK reads both env vars on every request. A child that has had its env tampered with by another process in the same session can send arbitrary requests; this is acceptable because the supervisor's constant-time token check (Phase 11 D-01) is the actual auth gate and the per-type mask validator (Plan 18-01 D-03 + D-07) is the actual policy gate. The SDK does not log either env value. |
| SDK → SupervisorSocket transport | The SDK takes a `&mut SupervisorSocket` from the caller, NOT opening the connection itself. This means the caller controls the pipe lifecycle; misuse (e.g. holding the pipe across a fork) is the caller's responsibility. The Phase 11 transport itself is the trusted side (named pipe with SDDL `S:(ML;;NW;;;LW)` from `bind_low_integrity`). |
| SDK reconstructed RawSocket → caller's stack | `reconstruct_socket_from_blob` returns a freshly-reconstructed SOCKET from the WSAPROTOCOL_INFOW blob via WSASocketW. The caller now owns the SOCKET and is responsible for `closesocket`. The SDK does not log the blob bytes or the resulting SOCKET value. |

Acceptance criteria grep-asserts the `pub use` line contains exactly the 8 named symbols. |
| T-18-04-07 | D (Denial of Service) | SDK call blocks indefinitely on `recv_response` if supervisor is hung | accept (caller responsibility) | The `SupervisorSocket::recv_response` is the Phase 11 transport; setting a read timeout is the caller's responsibility (per the `ApprovalBackend` trait doc-comment in `mod.rs:90-92` — "supervisor should set a read timeout on the socket to prevent indefinite blocking"). | T-18-04-09 | I (Information Disclosure) | Environment-variable side-channel: `NONO_SESSION_ID` leaks the supervisor session ID | accept | `session_id` is not security-sensitive per Phase 11 D-01 / D-11 (only `session_token` is the auth credential). A caller that doesn't want to send an AIPC request doesn't call those methods. Tests use the `with_test_session_token` save/restore helper to scope env mutation. CLAUDE.md guidance on env-mutating tests (§ "Environment variables in tests") is the governing standard. |
