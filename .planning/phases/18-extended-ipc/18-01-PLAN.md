```rust
pub fn broker_file_handle_to_process(
    file: &File,
    target_process: BrokerTargetProcess,
    access: crate::capability::AccessMode,
) -> Result<ResourceGrant> {
    let mut duplicated: HANDLE = std::ptr::null_mut();
    let source_handle = file.as_raw_handle() as HANDLE;
    // SAFETY: `source_handle` comes from a live `File`, the current process is the
    // source process, `target_process` was wrapped by the caller from a live process
    // handle, and `duplicated` points to writable storage.
    let ok = unsafe {
        DuplicateHandle(
            GetCurrentProcess(), source_handle, target_process.raw(),
            &mut duplicated, 0, 0, DUPLICATE_SAME_ACCESS,
        )
    };
    if ok == 0 || duplicated.is_null() {
        return Err(NonoError::SandboxInit(format!(
            "Failed to duplicate Windows handle into supervised child: {}",
            std::io::Error::last_os_error()
        )));
    }
    Ok(ResourceGrant::duplicated_windows_file_handle(
        duplicated as usize as u64, access,
    ))
}
```

       pub fn broker_event_to_process(
           handle: HANDLE,
           target_process: BrokerTargetProcess,
           mask: u32,
       ) -> Result<ResourceGrant> {
           let mut duplicated: HANDLE = std::ptr::null_mut();

           // SAFETY: `event` is a live HANDLE returned by CreateEventW above.
           unsafe { CloseHandle(event); }
           grant.map(Some)
       }

       fn validate_aipc_object_name(name: &str) -> Result<(), nono::NonoError> {
           if name.is_empty() || name.len() > 64 {
               return Err(nono::NonoError::SandboxInit(format!(
                   "AIPC object name must be 1..=64 bytes (got {})", name.len()
               )));
           }
           for ch in name.chars() {
               if ch == '\\' || ch == '/' || ch == ':' || ch == '\0' || ch.is_control() {
                   return Err(nono::NonoError::SandboxInit(format!(
                       "AIPC object name contains forbidden char {ch:?}"
                   )));
               }
           }
           Ok(())
       }
